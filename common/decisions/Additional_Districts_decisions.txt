	owned_planets_only = yes
	enactment_time = 0

		category = decisions

			minerals = 1000
			alloys = 500
			exotic_gases = 100
		}
	}

		NOT = {
			has_planet_flag = do_ad_undo_display
		}

			has_planet_flag = do_ad_sky_cities
		}

	}

		NOT = {
