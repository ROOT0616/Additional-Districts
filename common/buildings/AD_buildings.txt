# building_ad_Palace  宮殿			安定度+10						側近Job+1			(国家元首や君主が住み、執務を行う宮殿。彼らの側近やスタッフは国家運営を大きく左右する。)企業以外	維持:エネ3  			建設:鉱物1000、時間550
building_ad_Palace = {
	base_buildtime = 550
	base_cap_amount = 1
	category = government
	potential = {
		owner = {
			OR = {
				is_regular_empire = yes
				AND = {
					has_origin = origin_fear_of_the_dark
					is_country_type = primitive
				}
			}
			NOR = {
				has_authority = auth_corporate
				has_country_flag = build_ad_Palace
			}
		}
		NOT = { has_modifier = slave_colony }
	}
	planet_modifier = {
		planet_stability_add = 10
		job_ad_ministor_add = 1
	}
	resources = {
		category = planet_buildings
		cost = {
			minerals = 1000
		}
		upkeep = {
			energy = 3
		}
	}
	on_built = {
		owner = {
			if = {
				limit = {
					NOT = {
						has_country_flag = build_ad_Palace
					}
				}
				set_country_flag = build_ad_Palace
			}
		}
		planet = {
			set_planet_flag = build_planet_ad_Palace
		}
	}
	on_destroy = {
		owner = {
			remove_country_flag = build_ad_Palace
		}
	}
	on_queued = {
		owner = {
			set_country_flag = build_ad_Palace
		}
	}
	on_unqueued = {
		owner = {
			if = {
				limit = {
					has_country_flag = build_ad_Palace
				}
				remove_country_flag = build_ad_Palace
			}
		}
	}
	ai_resource_production = {
		trigger = { 
			always = yes
		}
		influence = 1
	}
}
